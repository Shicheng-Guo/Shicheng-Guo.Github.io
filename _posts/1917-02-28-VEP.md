```

perl -MCPAN -Mlocal::lib -e 'CPAN::install(Archive::Zip)'
perl -MCPAN -Mlocal::lib -e 'CPAN::install(DBI)'
perl -MCPAN -Mlocal::lib -e 'CPAN::install(Try::Tiny)'

git clone https://github.com/Ensembl/ensembl-vep.git
cd ensembl-vep
perl INSTALL.pl
```
